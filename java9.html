21 Полный код 
let car = {
    brand: "Toyota",
    model: "Corolla",
    year: 2015
};
car.year = new Date().getFullYear();
car.color = "red";
delete car.model;
console.log("brand" in car); // true
for (let key in car) {
    console.log(`${key}: ${car[key]}`);
}
let carCopy = { ...car };
carCopy.color = "blue";
console.log("Оригинал:", car);
console.log("Копия:", carCopy);
car.owner = {
    name: "John",
    age: 30
};
const { owner: { name } } = car;
console.log("Имя владельца:", name);
console.log("Ключи объекта:", Object.keys(car));
console.log("Значения объекта:", Object.values(car));
console.log("Ключи и значения объекта:", Object.entries(car));
