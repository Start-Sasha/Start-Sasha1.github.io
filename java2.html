1 Описание работы: В данной лабораторной работе мы создадим HTML-страницу, на которой будут размещены счетчики и кнопки для их изменения. Мы будем использовать JavaScript для обработки событий нажатия кнопок и изменения значений счетчиков. 
<!DOCTYPE html>
<html>
<body>
    <span id="counter1">0</span>
    <button onclick="incrementCounter(1)">+</button>
    <div id="counter2">0</div>
    <button onclick="decrementCounter(2)">-</button>
    <p id="counter3">0</p>
    <button onclick="incrementCounter(3)">+</button>
    <h1 id="counter4">0</h1>
    <button onclick="decrementCounter(4)">-</button>

    <script>
        function incrementCounter(counterNumber) {
            let counterElement = document.getElementById(`counter${counterNumber}`);
            let currentValue = parseInt(counterElement.innerHTML);
            counterElement.innerHTML = currentValue + 1;
        }

        function decrementCounter(counterNumber) {
            let counterElement = document.getElementById(`counter${counterNumber}`);
            let currentValue = parseInt(counterElement.innerHTML);
            counterElement.innerHTML = currentValue - 1;
        }
    </script>
</body>
</html>

<!DOCTYPE html>
<html>
<body>
    <span id="count">100</span>
    <button onclick="Counter()">+</button>

    <script>
        function Counter() {
            let counterElement = document.getElementById("count");
            let currentValue = parseInt(counterElement.innerHTML);
            counterElement.innerHTML = currentValue + 1;
        }
    </script>
