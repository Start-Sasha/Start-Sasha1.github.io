let createGradeScale = score => {
    let scales = [
        { min: 90, grade: "5+", message: "Превосходно!" },
        { min: 80, grade: "5", message: "Отлично!" },
        { min: 70, grade: "4", message: "Хорошо!" },
        { min: 60, grade: "3", message: "Удовлетворительно" },
        { min: 0, grade: "2", message: "Нужно больше практики" }
    ];

    // Находим подходящую шкалу
    let result = scales.find(scale => score >= scale.min);

    // Проверяем, найдена ли шкала
    if (result) {
        alert(`Оценка: ${result.grade}\nСообщение: ${result.message}`);
    } else {
        alert("Невозможно определить оценку для данного балла.");
    }
};

let factorial = (n, isTopLevel = true) => {
    if (n <= 1) return 1; // Базовый случай: факториал 0 или 1 равен 1
    const result = n * factorial(n - 1, false); // Рекурсивный случай
    if (isTopLevel) {
        alert(result); // Выводим результат только на верхнем уровне рекурсии
    }
    return result;
};

factorial(5)
